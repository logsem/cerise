  Lemma regname_decidable (r1 r2 : RegName) :
    Decidable.decidable (r1 = r2).
    intros.
    unfold Decidable.decidable.
    destruct (decide (r1 = r2)) ; [by left |by right].
  Unshelve. Fail idtac. Admitted.

  Lemma list_remove_elem_cons
    `{A : Type} `{EqDecision A} (a : A) (la : list A) :
    a ∉ la ->
    (list_remove_elem a (a::la)) = (list_remove_elem a la).
  Proof.
    intros Hnotin.
    apply list_remove_elem_notin in Hnotin.
    assert (a ∈ a::la) as Hin by set_solver.
    apply list_remove_elem_in in Hin.
    destruct Hin as (la' & Hla' & Hin).
    rewrite Hnotin Hla'.
    cbn in Hin.
    rewrite decide_True /= in Hin; [|done].
    by inversion Hin.
  Unshelve. Fail idtac. Admitted.

  Lemma list_remove_elem_idem
    `{A : Type} `{EqDecision A} (a : A) (la : list A) :
    NoDup la ->
    (list_remove_elem a (list_remove_elem a la)) =
    (list_remove_elem a la).
  Proof.
    intros HNoDup.
    rewrite list_remove_elem_notin; first done.
    by apply not_elemof_list_remove_elem.
  Unshelve. Fail idtac. Admitted.

  Lemma list_remove_elem_neq
    `{A : Type} `{EqDecision A} (a a' : A) (la : list A) :
    NoDup la ->
    a' ∈ la ->
    a ∈ list_remove_elem a' la ->
    a ≠ a'.
  Proof.
    intros HnoDup Hin Hrem.
    pose proof (Hin' := Hin).
    apply list_remove_elem_in in Hin.
    destruct Hin as (la' & Hla' & Hrem'); subst.
    apply not_elemof_list_remove_list in Hrem'; auto.
    set_solver.
  Unshelve. Fail idtac. Admitted.

  Definition allow_sweep_mask
    (la : list Addr) (v : Version) (pc_a : Addr) (pc_v : Version) : coPset :=
    compute_mask_region (⊤ ∖ ↑logN.@(pc_a, pc_v)) (list_remove_elem pc_a la) v.

  Lemma allow_sweep_mask_remove_pca
    (la : list Addr) (v : Version) (pc_a : Addr) (pc_v : Version) :
    NoDup la ->
    allow_sweep_mask (list_remove_elem pc_a la) v pc_a pc_v =
    allow_sweep_mask la v pc_a pc_v.
  Proof.
    intros HNoDup.
    by rewrite /allow_sweep_mask list_remove_elem_idem.
  Unshelve. Fail idtac. Admitted.

  Definition region_open_resources
    (pc_a : Addr) (pc_v : Version)
    (la : list Addr) (v : Version)
    (lws : list LWord) (Ps : list D) (b : bool)
    : iProp Σ :=

    let E := ⊤ ∖ ↑logN.@(pc_a, pc_v) in
    let E' := allow_sweep_mask la v pc_a pc_v in

    ([∗ list] lw;Pw ∈ lws;Ps, (if b then ▷ (Pw : D) lw else (Pw : D) lw))
    (* ∗ ( ⌜ Persistent (∗ list lw;Pw ∈ lws;Ps, (Pw : D) lw) ⌝ ) *)
      ∗ ([∗ list] Pa ∈ Ps, read_cond Pa interp)
      ∗ ( (▷ ([∗ list] a_Pa ∈ zip la Ps, (interp_ref_inv a_Pa.1 v a_Pa.2)))
         ={E', E }=∗ True).

  Definition reg_allows_sweep (lregs : LReg) (r : RegName) p b e a v :=
    lregs !! r = Some (LCap p b e a v) ∧ readAllowed p = true.

  (* Description of what the resources are supposed to look like after opening the region *)
  (*    if we need to, but before closing the region up again*)
  Definition allow_sweep_res
    (lregs : LReg) (r : RegName)
    (pc_a : Addr) (pc_v : Version)
    (p : Perm) (b e a : Addr)
    (v : Version) (Ps : list D)
    :=

    let la := (list_remove_elem pc_a (finz.seq_between b e)) in
    let E := ⊤ ∖ ↑logN.@(pc_a, pc_v) in
    let E' := allow_sweep_mask la v pc_a pc_v in
    let cond_open :=
      decide (reg_allows_sweep lregs r p b e a v /\ (r = PC \/ pc_a ∉ (finz.seq_between b e))) in

    (⌜read_reg_inr lregs r p b e a v⌝ ∗
     if cond_open
     then
      (|={E, E'}=>
         ∃ (lws :list LWord),
         ⌜ length lws = length la ⌝
         ∗ ([∗ map] la↦lw ∈ (logical_region_map la lws v), la ↦ₐ lw)
         ∗ region_open_resources pc_a pc_v la v lws Ps true)%I
     else True)%I.

  Lemma create_sweep_res:
    ∀ (lregs : leibnizO LReg)
      (pc_p : Perm) (pc_b pc_e pc_a : Addr) (pc_v : Version)
      (src : RegName) (p : Perm) (b e a : Addr) (v : Version),
    let la := (list_remove_elem pc_a (finz.seq_between b e)) in
    read_reg_inr (<[PC:=LCap pc_p pc_b pc_e pc_a pc_v]> lregs) src p b e a v
    → (∀ (r : RegName) lw, ⌜r ≠ PC⌝ → ⌜lregs !! r = Some lw⌝ → (fixpoint interp1) lw)
      -∗ interp (LCap pc_p pc_b pc_e pc_a pc_v)
         -∗ ⌜ pc_a ∈ finz.seq_between pc_b pc_e ⌝
            -∗ (∃ (Ps : list D),
                   ⌜ length la = length Ps ⌝
                   ∗ allow_sweep_res
                       (<[PC:=LCap pc_p pc_b pc_e pc_a pc_v]> lregs) src
                       pc_a pc_v
                       p b e a v Ps).
  Proof.
    intros * Hread.
    iIntros "#Hreg #Hinterp_pc %Hpca_inbounds".
    apply list_remove_elem_in in Hpca_inbounds.
    destruct Hpca_inbounds as (la' & <- & Hpca_inbounds).
    rewrite /allow_sweep_res.
    case_decide as Hallows; cycle 1.
    { iExists (repeat (λne _, True%I) (length la)); iFrame "%".
      by rewrite repeat_length.
    }
    iFrame "%".
    cbn in Hpca_inbounds.
    apply bind_Some in Hpca_inbounds.
    destruct Hpca_inbounds as (? & Hrem & ?) ; simplify_eq.
    rewrite /reg_allows_sweep in Hallows.
    destruct Hallows as ( (Hreg & HreadAllowed) & Hdecide).
    assert (la ⊆+ finz.seq_between b e) as Hla_subseteq by apply list_remove_elem_submsteq.
    assert (NoDup la) as Hla_NoDup by (by apply list_remove_elem_NoDup).

    destruct (decide (src = PC)) as [-> | Hneq_src_pc].
    (* src = PC *)
    - rewrite /read_reg_inr in Hread; simplify_map_eq.
      rewrite /allow_sweep_mask.
      rewrite list_remove_elem_idem; last done.
      iDestruct (interp_open_region $ (⊤ ∖ ↑logN.@(a, v)) with "Hinterp_pc")
        as (Ps) "[%Hlen_Ps Hmod]" ; eauto.
      { eapply Forall_forall. intros a' Ha'.
        subst la.
        eapply namespaces.coPset_subseteq_difference_r; auto.
        assert (a' ≠ a).
        {
          eapply list_remove_elem_neq; cycle 1 ; eauto.
          apply list_remove_Some in Hrem.
          setoid_rewrite Hrem; set_solver.
        }
        solve_ndisj.
      }
      iExists Ps.
      iSplit ; first by subst.
      iMod "Hmod".
      iDestruct "Hmod" as (lws) "(%Hlws & %Hpers & Hmem & HPs & HPs' & Hclose)".
      iExists lws.
      iFrame; iFrame "%".
      iModIntro.
      iIntros "H".
      iDestruct ("Hclose" with "H") as "Hclose".
      rewrite /allow_sweep_mask.
      rewrite list_remove_elem_notin.
      subst la.
      iFrame.
      apply not_elemof_list_remove_elem; done.

    (* src ≠ PC *)
    - destruct Hdecide as [Hcontra|Ha_notin_rem] ; first contradiction.
      rewrite /read_reg_inr in Hread; simplify_map_eq.
      iAssert (interp (LCap p b e a v)) as "#Hinterp_src" ; first (iApply "Hreg"; eauto).
      iDestruct (interp_open_region $ (⊤ ∖ ↑logN.@(pc_a, pc_v)) with "Hinterp_src")
        as (Ps) "[%Hlen_Ps Hmod]" ; eauto.
      { apply Forall_forall; intros a' Ha'.
        subst la.
        assert (a' ≠ pc_a).
        { intro ; subst. by apply not_elemof_list_remove_elem in Ha'. }
        apply namespaces.coPset_subseteq_difference_r; [solve_ndisj|set_solver].
      }
      iExists Ps.
      iSplit ; first by subst.
      iMod "Hmod".
      rewrite allow_sweep_mask_remove_pca; auto.
      iDestruct "Hmod" as (lws) "(%Hlws & %Hpers & Hmem & HPs & HPs' & Hclose)".
      iExists lws.
      iFrame; iFrame "%".
      iModIntro.
      iIntros "H".
      iDestruct ("Hclose" with "H") as "Hclose".
      by rewrite allow_sweep_mask_remove_pca.
  Unshelve. Fail idtac. Admitted.

  Definition allow_sweep_mem
    (lregs : LReg) (r : RegName)
    (pc_a : Addr) (pc_v : Version) (pc_lw : LWord)
    (p : Perm) (b e a : Addr) (v : Version)
    (lmem : LMem) (Ps : list D) :=

    let la := (list_remove_elem pc_a (finz.seq_between b e)) in
    let cond_open :=
     decide (reg_allows_sweep lregs r p b e a v /\ (r = PC \/ pc_a ∉ (finz.seq_between b e)))
    in

    (⌜read_reg_inr lregs r p b e a v⌝ ∗
     if cond_open
     then (∃ (lws : list LWord),
              ⌜length lws = length la⌝
              ∗ ⌜lmem = <[(pc_a, pc_v):=pc_lw]> (logical_region_map la lws v)⌝
              ∗ region_open_resources pc_a pc_v la v lws Ps true)
     else ⌜lmem = <[(pc_a, pc_v):=pc_lw]> ∅⌝)%I.

  Lemma sweep_res_implies_mem_map:
    ∀ (lregs : leibnizO LReg)
      (pc_a : Addr) (pc_v : Version) (pc_lw : LWord) (r : RegName)
      (p : Perm) (b e a : Addr) (v : Version) (Ps : list D),

    let la := (list_remove_elem pc_a (finz.seq_between b e)) in
    let E := ⊤ ∖ ↑logN.@(pc_a, pc_v) in
    let E' := allow_sweep_mask la v pc_a pc_v in
    let cond_open :=
     decide (reg_allows_sweep lregs r p b e a v /\ (r = PC \/ pc_a ∉ (finz.seq_between b e)))
    in

    allow_sweep_res lregs r pc_a pc_v p b e a v Ps
    -∗ (pc_a, pc_v) ↦ₐ pc_lw
    ={ E ,if cond_open then E' else E}=∗
    ∃ lmem : LMem,
      allow_sweep_mem lregs r pc_a pc_v pc_lw p b e a v lmem Ps
      ∗ ▷ ([∗ map] la↦lw ∈ lmem, la ↦ₐ lw).
  Admitted.

  (* TODO @June redo the proof in the same style as Load or Store *)
  Lemma isunique_case (lregs : leibnizO LReg)
    (ppc : Perm) (bpc epc apc : Addr) (vpc : Version)
    (lwpc : LWord) (dst src : RegName) (P : D):
    ftlr_instr lregs ppc bpc epc apc vpc lwpc (IsUnique dst src) P.
  Proof.
    intros Hp Hsome i Hbae Hi.
    iIntros "#IH #Hinv #Hinva #Hreg #(Hread & Hwrite & %HpersP) Hown Ha HP Hcls HPC Hmap".
    specialize (HpersP lwpc).
    rewrite delete_insert_delete.
    iDestruct ((big_sepM_delete _ _ PC) with "[HPC Hmap]") as "Hmap /=";
      [apply lookup_insert|rewrite delete_insert_delete;iFrame|]. simpl.

    (* To read out PC's name later, and needed when calling wp_load *)
    assert(∀ x : RegName, is_Some (<[PC:=LCap ppc bpc epc apc vpc]> lregs !! x)) as Hsome'.
    {
      intros. destruct (decide (x = PC)) as [Hx|Hx].
      rewrite Hx lookup_insert; unfold is_Some. by eexists.
      by rewrite lookup_insert_ne.
    }

    (* Initializing the names for the values of Hsrc now, to instantiate the existentials in step 1 *)
    assert (∃ psrc bsrc esrc asrc vsrc, read_reg_inr (<[PC:=LCap ppc bpc epc apc vpc]> lregs) src psrc bsrc esrc asrc vsrc)
      as (psrc & bsrc & esrc & asrc & vsrc & HVsrc).
    {
      specialize Hsome' with src as Hsrc.
      destruct Hsrc as [wsrc Hsomesrc].
      unfold read_reg_inr. rewrite Hsomesrc.
      destruct wsrc as [|[ psrc bsrc esrc asrc vsrc|] | ]; try done.
      by repeat eexists.
    }

    (* Step 1: open the region, if necessary,
       and store all the resources obtained from the region in allow_load_res *)
    iDestruct (create_sweep_res with "[$Hreg] [$Hinv] [%]") as (Ps) "[%HlenPs HSweepRes]"
    ; [ eassumption
      | by apply elem_of_finz_seq_between
      |].

    (* Set Nested Proofs Allowed. *)
    (* rewrite /allow_sweep_res. *)

    (* Step 2: derive the concrete map of memory we need, and any spatial predicates holding over it *)
    iMod (sweep_res_implies_mem_map with "HSweepRes Ha") as (mem) "[HSweepMem >HMemRes]".
    set (E := ⊤ ∖ ↑logN.@(apc, vpc)).
    set (E' := allow_sweep_mask (list_remove_elem apc (finz.seq_between bsrc esrc)) vsrc apc vpc).
    rewrite /allow_sweep_mem.
    set (cond_open :=
           decide
